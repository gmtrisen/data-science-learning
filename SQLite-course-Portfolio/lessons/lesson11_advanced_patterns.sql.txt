-- Common Table Expression (CTE)

WITH dept_ranked AS (
SELECT
e.first_name,
e.last_name,
d.department_name,
e.salary,
RANK() OVER(
PARTITION BY d.department_name
ORDER BY e.salary DESC
) AS dept_rank
FROM employees e
JOIN departments d
ON e.department_id = d.department_id
)

SELECT *
FROM dept_ranked
WHERE dept_rank = 1;

-- EXISTS example

SELECT department_name
FROM departments d
WHERE EXISTS(
SELECT 1
FROM employees e
WHERE e.department_id = d.department_id
);
