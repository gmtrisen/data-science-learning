-- LESSON 9: Database Normalization

-- Create departments table

CREATE TABLE departments_normalized(
    department_id INTEGER PRIMARY KEY,
    department_name TEXT UNIQUE,
    manager TEXT
);

-- Create employees table linked to departments

CREATE TABLE employees_normalized(
    employee_id INTEGER PRIMARY KEY,
    first_name TEXT,
    last_name TEXT,
    salary REAL,
    department_id INTEGER,
    FOREIGN KEY (department_id)
    REFERENCES departments_normalized(department_id)
);

-- Join normalized tables

SELECT
e.first_name,
e.last_name,
d.department_name
FROM employees_normalized e
JOIN departments_normalized d
ON e.department_id = d.department_id;
